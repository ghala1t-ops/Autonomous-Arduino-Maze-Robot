#include <NewPing.h>
#include <Servo.h>
#include <SD.h>
#include <SPI.h>
#include <TinyGPS++.h>
#include <SoftwareSerial.h>
#include <Wire.h>
#include <MPU6050.h>
#include "HUSKYLENS.h"

// -------------------------
// HuskyLens
// -------------------------
HUSKYLENS huskylens;

// -------------------------
// Pins configuration
// -------------------------
#define TRIG_PIN A0
#define ECHO_PIN A1
#define MAX_DISTANCE 300  // Max distance for ultrasonic in cm

#define IN1 7  // Motor A
#define IN2 6
#define IN3 5  // Motor B
#define IN4 4

// MPU6050
MPU6050 mpu;
int16_t ax, ay, az, gx, gy, gz;

// Ultrasonic + Servo
NewPing sonar(TRIG_PIN, ECHO_PIN, MAX_DISTANCE);
Servo myservo;
int distance = 0;

// -------------------------
// GPS + SD
// -------------------------
File myFile;
static const int RXPin = 8, TXPin = 9;

TinyGPSPlus gps;
SoftwareSerial SerialGPS(RXPin, TXPin);

// Logging timer
unsigned long lastLogTime = 0;
const unsigned long LOG_INTERVAL_MS = 500;  // Periodic log interval

// --------------------------------------------------
// Convert HuskyLens ID → readable object name
// --------------------------------------------------
String getObstacleNameFromID(int id) {
  switch (id) {
    case 1: return "Wall";
    case 2: return "Box";
    case 3: return "Human";
    // Add more IDs according to your HuskyLens training
    default: return "Unknown object";
  }
}

// --------------------------------------------------
// Read object detected by HuskyLens
// --------------------------------------------------
String detectObstacleWithHusky() {

  if (!huskylens.request())
    return "No response from HuskyLens";

  if (!huskylens.isLearned())
    return "No learned objects";

  if (!huskylens.available())
    return "No object detected";

  HUSKYLENSResult result = huskylens.read();
  int id = result.ID;
  String name = getObstacleNameFromID(id);

  return "ID: " + String(id) + " -> " + name;
}

// --------------------------------------------------
// Generic logger: GPS + ACC + obstacle + distance
// obstacleInfo: description or "None"
// frontDistance: distance in cm, or -1 if not applicable
// --------------------------------------------------
void logData(const String& obstacleInfo, int frontDistance) {

  // Update GPS stream
  while (SerialGPS.available() > 0)
    gps.encode(SerialGPS.read());

  // If GPS not ready, we skip logging to avoid invalid coordinates
  if (!gps.location.isValid()) {
    Serial.println("GPS invalid, skipping log");
    return;
  }

  // GPS
  String lat = String(gps.location.lat(), 6);
  String lng = String(gps.location.lng(), 6);

  // Accelerometer readings
  mpu.getMotion6(&ax, &ay, &az, &gx, &gy, &gz);
  String accX = String(ax / 16384.0, 3);
  String accY = String(ay / 16384.0, 3);
  String accZ = String(az / 16384.0, 3);

  // Build CSV line
  String line = lat + "," + lng + "," +
                accX + "," + accY + "," + accZ + "," +
                "\"" + obstacleInfo + "\"," + String(frontDistance);

  myFile = SD.open("Log.csv", FILE_WRITE);
  if (myFile) {
    myFile.println(line);
    myFile.close();
    Serial.println("Log Saved: " + line);
  } else {
    Serial.println("Failed to open Log.csv");
  }
}

// --------------------------------------------------
// Setup
// --------------------------------------------------
void setup() {
  Serial.begin(9600);

  myservo.attach(3);
  myservo.write(90);

  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT);
  pinMode(IN4, OUTPUT);

  // GPS
  SerialGPS.begin(9600);

  // SD Card init
  Serial.println("Initializing SD card...");
  if (!SD.begin(10)) {
    Serial.println("SD init failed!");
  } else {
    Serial.println("SD init OK");

    // Create unified log file
    myFile = SD.open("Log.csv", FILE_WRITE);
    if (myFile) {
      myFile.println("Latitude,Longitude,AccX,AccY,AccZ,Obstacle,Distance_cm");
      myFile.close();
    } else {
      Serial.println("Error creating Log.csv");
    }
  }

  // I2C Bus
  Wire.begin();

  // MPU6050
  mpu.initialize();
  Serial.println(mpu.testConnection() ? "MPU6050 OK" : "MPU6050 FAILED");

  // HuskyLens
  if (!huskylens.begin(Wire)) {
    Serial.println("HUSKYLENS FAILED! Check wiring.");
  } else {
    Serial.println("HUSKYLENS OK");
  }
}

// --------------------------------------------------
// Motor Control
// --------------------------------------------------
void forward() {
  digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW);
  digitalWrite(IN3, HIGH); digitalWrite(IN4, LOW);
}

void backward() {
  digitalWrite(IN1, LOW);  digitalWrite(IN2, HIGH);
  digitalWrite(IN3, LOW);  digitalWrite(IN4, HIGH);
}

void turnRight() {
  digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW);  digitalWrite(IN4, HIGH);
  delay(250);
  stopRobot();
}

void turnLeft() {
  digitalWrite(IN1, LOW);  digitalWrite(IN2, HIGH);
  digitalWrite(IN3, HIGH); digitalWrite(IN4, LOW);
  delay(250);
  stopRobot();
}

void stopRobot() {
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW);
  digitalWrite(IN4, LOW);
}

// --------------------------------------------------
// Ultrasonic distance
// --------------------------------------------------
int getDistance() {
  delay(50);
  int cm = sonar.ping_cm();
  return (cm == 0) ? 300 : cm;  // 0 means no echo → treat as far
}

// --------------------------------------------------
// Main loop
// --------------------------------------------------
void loop() {

  myservo.write(90);  // Look forward
  delay(300);

  distance = getDistance();
  Serial.print("Front Distance: ");
  Serial.println(distance);

  if (distance > 50) {
    // No close obstacle → move forward
    forward();
  } else {
    // Obstacle detected by ultrasonic
    stopRobot();
    delay(300);

    // Ask HuskyLens what it sees
    String obstacleInfo = detectObstacleWithHusky();
    Serial.println("Husky Detected: " + obstacleInfo);

    // Log event with obstacle description and real distance
    logData(obstacleInfo, distance);

    // Scan right
    myservo.write(20);
    delay(400);
    int distanceRight = getDistance();

    // Scan left
    myservo.write(160);
    delay(400);
    int distanceLeft = getDistance();

    // Return center
    myservo.write(90);
    delay(200);

    // Choose direction with more free space
    if (distanceRight > distanceLeft) {
      turnRight();
      delay(500);
    } else {
      turnLeft();
      delay(500);
    }
  }

  // Periodic logging (GPS + ACC only, no obstacle)
  if (millis() - lastLogTime > LOG_INTERVAL_MS) {
    // "None" means no specific obstacle event at this moment
    // -1 means distance not relevant for this periodic log
    logData("None", -1);
    lastLogTime = millis();
  }
}
